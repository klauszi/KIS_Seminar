\section{Einführung}

\subsection{Motivation}

\begin{frame}{\insertsubsection}
  \begin{block}<+->{Messgeräte liefern unzuverlässige Daten}
    \begin{itemize}
      \item GPS Tracker sind unzuverlässig in der Nähe von Gebäuden
      \item Sensoren sind empfindlich gegenüber äußere Einflüsse
          \begin{itemize}
              \item Z.B. starker Fall der Temperaturen bei einem Windzug.
          \end{itemize}
    \end{itemize}
  \end{block}
\begin{figure}
    \includegraphics[width=\textwidth]{GPS_Motivation_Example.png}
    \caption{GPS-Tracking auf dem Campus der Tsinghua Universität \cite{Shaoxu17}}

\end{figure}

\end{frame}

\begin{frame}{\insertsubsection}
  \begin{block}<+->{Ansätze mit den Umgang von unzuverlässigen Daten}
    \begin{enumerate}
        \item Unzuverlässige Datenpunkte mit der Anomalienerkennung entfernen
            \begin{itemize}
                \item Einzelne Ausreißer werden entfernt  \neutranie
                \item Entfernen langer Folgen von Ausreißer machen Ergebnis unbrauchbar \frownie
                \item Lange Folgen von Ausreißer werden als solche ggf. nicht erkannt \frownie
            \end{itemize}
        \item Unzuverlässige Datenpunkte mit der Anomalienerkennung reparieren
            \begin{itemize}
                \item Einzelne Ausreißer werden korrigiert \smiley
                \item Lange Folgen von Ausreißer werden zu stark verändert (widerspricht den Minimum-Change-Prinzip) \neutranie
                \item Lange Folgen von Ausreißer werden als solche ggf. nicht verändert \frownie
            \end{itemize}
    \end{enumerate}
  \end{block}
\end{frame}

\begin{frame}{\insertsubsection}
    \begin{block}<+->{Hinzunahme von als wahr markierte Werte}
    \begin{enumerate}
        \item Markierung durch den Benutzer
            \begin{itemize}
            \item Z.B. markiert der Benutzer in unregelmäßigen Zeitabständen seinen aktuellen Standort
            \end{itemize}
        \item Präzise Messgeräte liefern in periodischen Zeitabstände korrekte Werte
    \end{enumerate}
  \end{block}

    \includegraphics[width=\textwidth]{markierte_GPS_Motivation_Example.png}
    \caption{GPS-Tracking mit markierten Werte auf dem Campus der Tsinghua Universität}
\end{figure}
\end{frame}



\subsection{Zielsetzung}

\begin{frame}{\insertsubsection}
  \begin{block}<+->{Ziel der Arbeit}
    \begin{itemize}
      \item Einfliessen der markierten Werte in die Anomalienerkennung
            \begin{itemize}
                \item lange Folgen von Ausreißer sollen identifiziert werden
            \end{itemize}
      \item Anomalienerkennung mit den Minimum-Change-Prinzip vereinbaren   
            \begin{itemize}
                \item keine drastische Veränderungen der Messwerte
            \end{itemize}
      \item Einfliessen der markierten Werte in die anderen State-of-Art Verfahren
      \item Neues Verfahren mit den anderen Verfahren empirisch vergleichen
    \end{itemize}
  \end{block}
\end{frame}

\section{Einführung}

\subsection{Motivation}

\begin{frame}{\insertsubsection}
  \begin{block}<+->{Messgeräte liefern unzuverlässige Daten}
    \begin{itemize}
      \item GPS Tracker sind nahe von Gebäuden unzuverlässig
      \item Sensoren sind empfindlich gegenüber äußere Einflüsse
          \begin{itemize}
              \item Z.B. starker Fall der Temperaturen bei einem Windzug
          \end{itemize}
    \end{itemize}
  \end{block}
\begin{figure}
    \includegraphics[width=\textwidth]{GPS_Motivation_Example.png}
    \caption{GPS-Tracking auf dem Campus der Tsinghua Universität \cite{Shaoxu17}}

\end{figure}

\end{frame}

\begin{frame}{\insertsubsection}
  \begin{block}<+->{Umgang von unzuverlässigen Daten mit Anomalienerkennung}
    \begin{enumerate}
        \item Unzuverlässige Datenpunkte entfernen
            \begin{itemize}
                \item Ausreißer werden entfernt  \smiley
                \item Entfernen aufeinanderfolgende Fehler machen Ergebnis unbrauchbar \textbf{oder}
                 werden als solche ggf. nicht entfernt \frownie
            \end{itemize}
        \item Unzuverlässige Datenpunkte reparieren
            \begin{itemize}
                \item Einzelne Ausreißer werden leicht korrigiert \neutranie
                \item Aufeinanderfolgende Fehler werden zu stark verändert (In der Praxis liegen die Messungen nahe bei den korrekten Werten) \neutranie
            \end{itemize}
    \end{enumerate}
  \end{block}
\end{frame}

\begin{frame}{\insertsubsection}
    \begin{block}<+->{Hinzunahme von korrekt markierten Werten}
    \begin{enumerate}
        \item Markierung durch den Benutzer
            \begin{itemize}
            \item Z.B. markiert der Benutzer in beliebigen Zeitabständen seinen aktuellen Standort
            \end{itemize}
        \item Präzise Messgeräte liefern in längeren Zeitabstände korrekte Werte
    \end{enumerate}
  \end{block}
    \includegraphics[width=\textwidth]{markierte_GPS_Motivation_Example.png}
\end{frame}



\subsection{Zielsetzung}

\begin{frame}{\insertsubsection}
  \begin{block}<+->{Ziel der Arbeit}
    \begin{enumerate}
      \item Berücksichtigung der markierten Werte in der Anomalienerkennung
            \begin{itemize}
                \item Aufeinanderfolgende Fehler sollen besser abgeschätzt werden
            \end{itemize}
      \item Anomalienreparatur mit den Minimum-Change-Prinzip vereinbaren   
            \begin{itemize}
                \item Keine drastische Veränderungen der Messwerte
            \end{itemize}
        \item Neue Anomalienreparatur hinsichtlich Berechnungslaufzeit, Ergebnisgenauigkeit usw. optimieren
      \item Neue Anomalienreparatur mit unterschiedlichen Einstellungen  mit den anderen Verfahren empirisch vergleichen
    \end{enumerate}
  \end{block}
\end{frame}
